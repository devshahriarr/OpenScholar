// USERS & ORG
Table roles {
  id int [pk, increment]
  name varchar(50) [unique, not null]
}

Table universities {
  id int [pk, increment]
  name text [not null]
}

Table departments {
  id int [pk, increment]
  university_id int
  name text [not null]
}

Table users {
  id uuid [pk]
  name text [not null]
  email varchar(255) [unique, not null]
  password_hash text [not null]
  role_id int [not null]
  university_id int
  department_id int
  status varchar(20) [default: 'active'] // active | suspended
  bio text
  profile_image_url text
  is_verified boolean [default: false]
  created_at timestamp [default: `now()`]
}


// PAPERS
Table categories {
  id int [pk, increment]
  name text [unique, not null]
}

Table papers {
  id uuid [pk]
  category_id int [not null]
  status varchar(20) [default: 'draft' , note: 'draft | pending | approved | rejected'] 
  comment_count int [default: 0]
  reaction_count int [default: 0]
  created_by uuid [not null]
  is_deleted boolean [default: false]
  created_at timestamp [default: `now()`]
}

Table paper_versions {
  id uuid [pk]
  paper_id uuid [not null]
  version_number int [not null]
  title text [not null]
  abstract text [not null]
  keywords text[] // use GIN index in DB
  pdf_url text [not null]
  is_published boolean [default: false]
  created_at timestamp [default: `now()`]

  Indexes {
    (paper_id, version_number) [unique]
  }
}


// AUTHORS
Table authors {
  id uuid [pk]
  name text [not null]
  normalized_name text
  user_id uuid

  Indexes {
    (user_id)
    (normalized_name)
  }
}

Table paper_authors {
  paper_version_id uuid
  author_id uuid
  author_order int [not null]

  Indexes {
    (paper_version_id, author_id) [pk]
    (paper_version_id, author_order) [unique]
  }
}


// TAGS
Table tags {
  id int [pk, increment]
  name varchar(100) [unique, not null]
}

Table paper_tags {
  paper_id uuid
  tag_id int

  Indexes {
    (paper_id, tag_id) [pk]
    (tag_id)
  }
}


// COMMENTS
Table comments {
  id uuid [pk]
  paper_id uuid
  user_id uuid
  content text [not null]
  parent_id uuid
  is_deleted boolean [default: false]
  created_at timestamp [default: `now()`]

  Indexes {
    (paper_id)
    (parent_id)
    (user_id)
  }
}


// REACTIONS
Table reactions {
  user_id uuid [not null]
  paper_id uuid [not null]
  type varchar(20) [note: 'like | love']
  created_at timestamp [default: `now()`]

  Indexes {
    (user_id, paper_id) [pk]
  }
}


// MODERATION
Table moderation_logs {
  id uuid [pk]
  paper_id uuid
  admin_id uuid
  action varchar(20) [note: 'approved | rejected']
  reason text
  created_at timestamp [default: `now()`]
}


// ANALYTICS
Table analytics_events {
  id bigint [pk, increment]
  paper_id uuid [not null]
  user_id uuid
  event_type varchar(20) [not null, note: 'view | download']
  created_at timestamp [default: `now()`]

  Indexes {
    (paper_id, event_type)
    (user_id)
  }
}

Table paper_metrics {
  paper_id uuid [pk]
  view_count int [default: 0]
  download_count int [default: 0]
  updated_at timestamp
}


// NOTIFICATIONS
Table notifications {
  id uuid [pk]
  user_id uuid
  type varchar(50)
  reference_id uuid
  message text
  is_read boolean [default: false]
  created_at timestamp [default: `now()`]

  Indexes {
    (user_id)
    (is_read)
    (created_at)
  }
}
// SAVED PAPERS
Table saved_papers {
  user_id uuid
  paper_id uuid
  created_at timestamp [default: `now()`]

  Indexes {
    (user_id, paper_id) [pk]
  }
}
// FOLLOWERS 
Table followers {
  follower_id uuid
  following_id uuid
  created_at timestamp [default: `now()`]

  Indexes {
    (follower_id, following_id) [pk]
    (following_id)
  }
}
